### 解构赋值的作用

#### 1.交换变量的值

```javascript
let x = 100;
let y = 200;
[y,x] = [x,y]; //x=200;y=100
```

#### 2.从函数返回多个值

	##### 2.1返回一个数组

```javascript
function fun() {
    return [1, 2, 3];
}
var [x, y, z] = fun();
console.log(x); //1
console.log(y); //2
console.log(z); //3
```

##### 2.2返回一个对象

```JavaScript
function fun() {
    return {
        id:'001',
        age:18
    }
}
var {id, age} = fun();
console.log(id);  //001
console.log(age); //18
```

```javascript
function fun() {
    return {
        id:'001',
        age:18
    }
}
var x = 5;
var {x:id, age} = fun();
console.log(x);  //001
```
